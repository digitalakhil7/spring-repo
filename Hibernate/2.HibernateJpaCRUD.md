4. Entity class
```java
@Data
@NoArgsConstructor
@AllArgsConstructor
@Entity
@Table(name = "emp")
public class Employee {
	@Id
	@Column(name = "eid")
	private Integer empId;
	@Column(name = "ename")
	private String empName;
	
	@Column(name = "ejd")
	@CreationTimestamp
	private LocalDateTime joiningDate;
	
	@Column(name = "eld")
	@UpdateTimestamp
	private LocalDateTime lastDate;
}
```

6. DAO Impl
```java
public class EmployeeDaoImpl implements EmployeeDao {

	EntityManagerFactory emf;
	
	public EmployeeDaoImpl() {
		emf = Persistence.createEntityManagerFactory("test");
	}
	
	@Override
	public Employee saveEmployee(Employee e) {
		EntityManager em = emf.createEntityManager();
		EntityTransaction t = em.getTransaction();
		t.begin();
		try {
			em.persist(e);
			t.commit();
		} catch (Exception e2) {
			t.rollback();
		}finally {
			em.close();
		}
		return e;
	}

	@Override
	public Employee findEmployeeById(Integer id) {
		EntityManager em = emf.createEntityManager();
		Employee find = em.find(Employee.class, id);
		em.close();
		return find;
	}

	@Override
	public Employee updateEmployeeById(Integer id, String empName) {
		EntityManager em = emf.createEntityManager();
		EntityTransaction t = em.getTransaction();
		Employee find = em.find(Employee.class, id);
		t.begin();
		if(find!=null)
		find.setEmpName(empName);
		t.commit();
		em.close();
		
		return find;
	}

	@Override
	public void deleteEmployeeById(Integer id) {
		EntityManager em = emf.createEntityManager();
		EntityTransaction t = em.getTransaction();
		Employee find = findEmployeeById(id);

		t.begin();
		if(find!=null) {
			em.remove(find);
			System.out.println("Employee removed with id: "+id);
		}else {
			System.out.println("Employee not found with id: "+id);
		}
		t.commit();
		em.close();
	}

}
```
