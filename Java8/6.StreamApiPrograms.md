public class Application {

	public static void main(String[] args) {

		l.add(new Employee(8, "Suresh", 31, "Male", "Development", 2015, 34500.0));
		l.add(new Employee(9, "Gita", 24, "Female", "Sales", 2016, 11500.0));
		l.add(new Employee(10, "Mahesh", 38, "Male", "Security", 2015, 11000.5));
		l.add(new Employee(11, "Gouri", 27, "Female", "Infrastructure", 2014, 15700.0));
		l.add(new Employee(12, "Nithin", 25, "Male", "Development", 2016, 28200.0));
		l.add(new Employee(13, "Swathi", 27, "Female", "Finance", 2013, 21300.0));
		l.add(new Employee(14, "Buttler", 24, "Male", "Sales", 2017, 10700.5));
		l.add(new Employee(15, "Ashok", 23, "Male", "Infrastructure", 2018, 12700.0));
		l.add(new Employee(16, "Sanvi", 26, "Female", "Development", 2015, 28900.0));
		
		// 1. How many male and female employees are there in the organization ?
		Map<String, Long> one = l.stream().collect(Collectors.groupingBy(Employee::getGender, Collectors.counting()));
		System.out.println(one);
		
		// 2. Print the name of all departments in the organization ?
		List<String> two = l.stream().map(Employee::getDepartment).distinct().collect(Collectors.toList());
		System.out.println(two);
		
		// 3. What is the average age of male and female employees ?
		Map<String, Double> three = l.stream().collect(Collectors.groupingBy(Employee::getGender, Collectors.averagingInt(Employee::getAge)));
		System.out.println(three);
		
		// 4. Get the details of highest paid employee in the organization ?
		Optional<Employee> four = l.stream().collect(Collectors.maxBy(Comparator.comparing(Employee::getSalary)));
		if(four.isPresent())
			System.out.println(four.get());

		// 5. Get the names of all employees who have joined after 2015 ?
		List<String> five = l.stream().filter(emp->emp.getYearOfJoining()>2015).map(emp->emp.getName()).collect(Collectors.toList());
		System.out.println(five);
		
		// 6. Count the number of employees in each department ?
		Map<String, Long> six = l.stream().collect(Collectors.groupingBy(Employee::getDepartment, Collectors.counting()));
		System.out.println(six);
	
		// 7. What is the average salary of each department ?
		Map<String, Double> seven = l.stream().collect(Collectors.groupingBy(Employee::getDepartment, Collectors.averagingDouble(Employee::getSalary)));
		System.out.println(seven);
	
		// 8. Get the details of youngest male employee in the Development department ?
		Optional<Employee> eight = l.stream().filter(emp->emp.getDepartment()=="Development").collect(Collectors.minBy(Comparator.comparing(Employee::getAge)));
		if(eight.isPresent())
			System.out.println(eight);
	
		// 9. Who has the most working experience in the organization ?
		Optional<Employee> nine = l.stream().collect(Collectors.minBy(Comparator.comparing(Employee::getYearOfJoining)));
		if(nine.isPresent())
			System.out.println(nine.get());
		
		// 10. How many male and female employees are there in the Sales team ?
		Map<String, Long> ten = l.stream().filter(emp->emp.getDepartment()=="Sales").collect(Collectors.groupingBy(Employee::getGender, Collectors.counting()));
		System.out.println(ten);
		
		// 11.  What is the average salary of male and female employees ?
		Map<String, Double> el = l.stream().collect(Collectors.groupingBy(Employee::getGender, Collectors.averagingDouble(Employee::getSalary)));
		System.out.println(el);
		
		// 12. List down the names of all employees in each department ?
		// https://javaconceptoftheday.com/solving-real-time-queries-using-java-8-features-employee-management-system/
		Map<String, List<Employee>> tw = l.stream().collect(Collectors.groupingBy(Employee::getDepartment));
		Set<Entry<String, List<Employee>>> set = tw.entrySet();
		for(Entry<String, List<Employee>> entry : set ) {
			System.out.println("-----------------------");
			System.out.println(entry.getKey());
			System.out.println("-----------------------");
			entry.getValue().forEach(System.out::println);
		}
		
		// 13.  What is the average salary and total salary of the whole organization ?
		DoubleSummaryStatistics th = l.stream().collect(Collectors.summarizingDouble(Employee::getSalary));
		System.out.println("Average Salary: "+th.getAverage());
		System.out.println("Total Salary: "+th.getSum());
	
		// 14.  Separate the employees who are younger or equal to 25 years from those employees who are older than 25 years ?
		// https://javaconceptoftheday.com/solving-real-time-queries-using-java-8-features-employee-management-system/
		Map<Boolean, List<Employee>> fo = l.stream().collect(Collectors.groupingBy(emp->emp.getAge()>25));
		Set<Entry<Boolean, List<Employee>>> entries = fo.entrySet();
		
		for(Entry<Boolean, List<Employee>> entry : entries) {
			if(entry.getKey()) {
			System.out.println("Employees more than 25 years");
			entry.getValue().forEach(System.out::println);
			}else {
				System.out.println("Employees less than or eq 25 years");
				entry.getValue().forEach(System.out::println);				
			}
			
		}
		
		// 15.  Who is the oldest employee in the organization?
		Optional<Employee> fi = l.stream().collect(Collectors.maxBy(Comparator.comparing(Employee::getAge)));
		if(fi.isPresent())
			System.out.println(fi.get());

		// 16. Employee with 2nd highesh salary (sory by salary, skip(1) and findFirst() )
		Optional<Employee> si = l.stream().sorted(Comparator.comparing(Employee::getSalary).reversed()).skip(1).findFirst();
		if(si.isPresent())
			System.out.println(si.get());

	}

}

				    }

				    System.out.println("----------------------------");

				    {
				        System.out.println(e.getName());
				    }
				};
