## WebClient CRUD
### Model class
```java
@Data
@NoArgsConstructor
@AllArgsConstructor
public class ToDo {
	private Integer postId;
	private Integer id;
	private String name;
	private String email;
	private String body;
}
```
### Runner
```java
@Component
public class MyRunner implements CommandLineRunner {

	@Override
	public void run(String... args) throws Exception {
		WebClient webClient = WebClient.create();
		
		
		ToDo todo = webClient
		.get()
		.uri("https://jsonplaceholder.typicode.com/comments/3")
		.retrieve()
		.bodyToMono(ToDo.class)
		.block();
		
		System.out.println(todo.getEmail());
		
		ToDo request = new ToDo(10, 10, "Akhil", "akhilved@gmail.com", "any body");
		
		ToDo response = webClient
		.post()
		.uri("https://jsonplaceholder.typicode.com/comments")
		.bodyValue(request)
		.retrieve()
		.bodyToMono(ToDo.class)
		.block();
		
		System.out.println(response.getEmail());
		
		ToDo requestPut = new ToDo(100, 200, "ChangedAkhil", "changedakhilved@gmail.com", "changed any body");
		
		ToDo responsePut = webClient
		.put()
		.uri("https://jsonplaceholder.typicode.com/comments/1")
		.bodyValue(requestPut)
		.retrieve()
		.bodyToMono(ToDo.class)
		.block();
		
		System.out.println(responsePut);
		
		ResponseEntity<Void> entity = webClient
		.delete()
		.uri("https://jsonplaceholder.typicode.com/comments/1")
		.retrieve()
		.toBodilessEntity()
		.block();
		
		System.out.println(entity.getStatusCode());
	}

}
```
